先根遍历

递归实现：

void preOrder(TreeNode *t){
    if(t != nullptr){
        visit(t);
        preOrder(t->left);
        preOrder(t->right);
    }
}

非递归实现：

void preOrder(treenode *t){
    treenode *p = t;
    initStack(s);
    if(t != nullptr) push(t, s);
    while(!isEmpty(s)){
        pop(s, p);
        visit(p);
        //一定要先右后左
        if(p->rchild != nullptr) push(p->rchild, s);
        if(p->lchild != nullptr) push(p->lchild, s);
    }
}

中根遍历

递归实现：

void midOrder(TreeNode *t){
    if(t != nullptr){
        midOrder(t->left);
        visit(t);
        midOrder(t->right);
    }
}

非递归实现：

void midOrder(treenode *t){
    initStack(s);
    treenode *p = t;
    while(p || !isEmpty(s)){
        if(p){
            push(p, s);
            p = p->lchild;//一直向左
        }
        else{
            pop(p, s);
            visit(p);
            p = p->rchild;
        }
    }
}

后根遍历

递归实现：

void postOrder(TreeNode *t){
    if(t != nullptr){
        postOrder(t->left);
        postOrder(t->right);
        visit(t);
    }
}

非递归实现：

void postOrder(treenode *t){
	initStack(s);
    treenode *p = t, *r = nullptr;
    while(p || !isEmpty(s)){
        if(p){
            push(p, s);
            p = p->lchild;//一直向左
        }
        else{
            pop(p, s);
            if(p->rchild && p->rchild != r){
                p = p->rchlid;
                push(p, s);
                p = p->lchild;
            }
            else{
                pop(p, s);
                visit(p);
                r = p;
                p = nullptr;
            }
        }
    }
}

层次遍历

void levelOrder(TreeNode *t){
    InitQueue(q);
    TreeNode *p;
    inQueue(q, t);
    while(!isEmpty(q)){
        deQueue(q, p);
        visit(p);
        if(p->left != nullptr) inQueue(q, p->left);
        if(p->right != nullptr) inQueue(q, p->right);
    }
}

树的高度

void depth(treenode *t){
    if(!t) return 0;
    int front = -1, rear = -1;
    int last = 0, level = 0;
    treenode q[MAXSIZE];
    q[++rear] = t;
    treenode *p;
    while(front < rear){
        p = Q[++front];
        if(p->lchild != nullptr) q[++rear] = p->lchild;
        if(p->rchild != nullptr) q[++rear] = p->rchild;
        if(front == last){
            level++;
            last = rear;
        }
    }
    return level;
}

树的宽度

typedef struct{
    treenode* data[MAXSIZE];
    int level[MAXSIZE];
    int front, rear;
}q;
int width(treenode *t){
    treenode *p;
    q.front = -1;
    q.rear = 0;
    q.data[rear] = t;
    q.level[rear] = 1;
    while(q.front < q.rear){
        q.front++;
        p = q.data[q.front];
        k = q.level[q.front];
        if(p->lchild != nullptr){
            q.rear++;
            q.data[q.rear] = p->lchild;
            q.level[q.rear] = k++;
        }
        if(p->rchild != nullptr){
            q.rear++;
            q.data[q.rear] = p->rchild;
            q.level[q.rear] = k++;
        }
    }
    int max = 0, i = 0, k = 1;
    while(i <= q.rear){
        int n = 0;
        if(i < q.rear && q.level == k) {
            n++;
            i++;
        }
        k = q.level[i];
        if(n > max) max = n;
    }
    return max;
}

由先序序列和中序序列建立二叉树

treenode* create(int *a, int* b, int l1, int h1, int l2, int h2){
    treenode *root = (treenode*)malloc(sizeof(treenode*));
    root->data = a[l1];
    for(int i= l2; b[i] != root->data; i++);
    llen = i - l2;
    rlen = h2 - i;
    if(llen) root-lchild = create(a, b, l1+1, l1+llen, l2, l2+llen-1);
    else root->lchild = nullptr;
    if(rlen) root-rchild = create(a, b, h1-rlen+1, h1, h2-rlen+1, h2);
    else root->rchild = nullptr;
    return root;
}
