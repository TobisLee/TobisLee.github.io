Leetcode 279. Perfect Squares

2019-12-13 2 min read # math # dp # leetcode # medium

Given a positive integer n, find the least number of perfect square numbers (for example, 1, 4, 9, 16, ...) which sum to n.

Example 1:

Input: n = 12
Output: 3 
Explanation: 12 = 4 + 4 + 4.

Example 2:

Input: n = 13
Output: 2
Explanation: 13 = 4 + 9.

Idea

  1. 使用dp，dp数组表示构成当前这个数的最少平方数的个数，状态转移方程：dp[i]=min(dp[i],dp[i−j∗j]+1)dp[i] = min(dp[i], dp[i-j*j]+1)dp[i]=min(dp[i],dp[i−j∗j]+1)
  2. 使用数学方法
    用数学理论可以证明这个问题只有四种答案，即1,2,3,4。
    由1个平方数构成的就是平方数
    由2个平方数构成的数的特点：减掉一个平方数，之后还是一个平方数
    由4个平方数构成的数应该满足4k(8∗m+7)4^k(8*m+7)4k(8∗m+7)。
    剩下的就是由3个平方数构成的数

Solution 1 --- dp

class Solution {
public:
    int numSquares(int n) {
        vector<int> dp(n+1, INT_MAX);
        dp[0] = 0;
        for (int i = 1; i <= n; i++) {
            for (int j = 1; j*j <= i; j++) {
                dp[i] = min(dp[i], dp[i - j*j]+1);
            }
        }
        return dp.back();
    }
};

Solution 2 --- math

class Solution {
private:
    bool isSquare(int n) {
        int sqrt_n = (int)sqrt(n);
        return (sqrt_n*sqrt_n == n);
    }
public:
    int numSquares(int n) {
        if (isSquare(n)) {return 1;}
        while ((n & 3) == 0) {
            // n%4 == 0
            n >>= 2;
        }
        if ((n & 7) == 7) {return 4;}
        int sqrt_n = (int)sqrt(n);
        for (int i =1; i <= sqrt_n; i++) {
            if (isSquare(n-i*i)) {return 2;}
        }
        return 3;
    }
};
