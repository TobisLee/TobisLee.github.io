删除数组中所有值为val的节点

void deleteItem(int *a, int val){
    int k = 0;
    for (int i = 0; i < a.length; i++){
        if(a[i] != val){
            a[k] = a[i];
            k++;
        }
    }
    a.length = k;
}

数组左移

  • 思路： ab -> a-1b -> a-1b-1 -> (a-1b-1)-1 = ba
void reverse(int *r, int from, int to){
    int i, temp;
    for (i = 0; i < (to-from+1)/2; i++) {
        temp = r[from + i];
        r[from+i] = r[to-i];
        r[to-i] = temp;
    }   
}
void converse(int *r, int len, int pos){
    reverse(r, 0, pos-1);
    reverse(r, pos, len-1);
    reverse(r, 0, len-1);
}
