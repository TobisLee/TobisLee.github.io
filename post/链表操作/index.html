获取链表中间节点（fast & slow node）

node* getMidNode(node* head){
    if(head == nullptr) return nullptr;
    node* fast = head;
    node* slow = head;
    while(fast->next != nullptr && fast->next->next != nullptr){
        slow = slow->next;
        fast = fast->next->next;
    }
    return slow;
}

获取链表倒数第K个值

node* getKNode(node* head, int k){
    if(k <= 0 || head == nullptr) return nullptr;
    node* p = head;
    node* q = head;
    while(k-- > 0) p = p->next;
    while(p->next != null){
        p = p->next;
        q = q->next;
    }
    return q;
}

删除链表中所有值为val的节点

node* deleteNode(node* head, int val){
    while(head != nullptr && head->val == val){
        head = head->next;
    }
    node* p = head;
    for(node* pre = head; head != nullptr; head = head->next){
        if(head->val == val) pre->next = p->next;
        else pre = p;
    }
    return head;
}

删除有序链表中的重复值

node* deleteDuplicate(node* head){
    if(head == nullptr) return nullptr;
    for(node* pre = head, *p = head->next; p != nullptr; p = p->next){
        if(pre->val == val) pre->next = p->next;
        else pre = p;
    }
    return head;
}

判断链表是否有环，有则返回入口点

node* detectCycle(node* head){
    node* slow = head;
    node* fast = head;
    while(fast != nullptr && fast->next != nullptr){
        slow = slow->next;
        fast = fast->next->next;
        if(slow == fast) break;
    }
    if(fast == nullptr || fast->next == nullptr) return nullptr;
    slow = head;
    while(slow != fast){
        slow = slow->next;
        fast = fast->next;
    }
    return slow;
}

将两个有序链表合并成一个链表

node* mergelist(node* a, node* b){
    if(a == null && b == null) return null;
    node* p1 = a;
    node* p2 = b;
    node* head = new node(-1);
    node* p = head;
    while(p1 != null && p2 != null){
        if(p1->val <= p2->val){
            p->next = p1;
            p = p1;
            p1 = p1->next;
        }
        else{
            p->next = p2;
            p = p2;
            p2 = p2->next;
        }
    }
    while(p1 != nullptr){
        p->next = p1;
        p = p1;
        p1 = p1->next;
    }
    while(p2 != nullptr){
        p->next = p2;
        p = p2;
        p2 = p2->next;
    }
    return head->next;
}
